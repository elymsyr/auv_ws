cmake_minimum_required(VERSION 3.8)
project(gazebo_sim)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)
find_package(gazebo_ros REQUIRED)
find_package(Eigen3 REQUIRED)
find_package(gazebo_msgs REQUIRED)

# === DEFINE AND BUILD YOUR C++ PLUGIN ===
add_library(auv_dynamics_plugin SHARED
  src/auv_dynamics_plugin.cpp
  src/vehicle_model.cpp
)

# ====================================================================
#          THIS IS THE MISSING BLOCK THAT FIXES THE ERROR
# ====================================================================
# This tells the compiler to look for headers in your 'include' directory
# when building the 'auv_dynamics_plugin' target.
target_include_directories(auv_dynamics_plugin
  PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)

# === LINK THE PLUGIN AGAINST DEPENDENCIES ===
ament_target_dependencies(auv_dynamics_plugin
  rclcpp
  std_msgs
  gazebo_ros
  gazebo_msgs
)

target_link_libraries(auv_dynamics_plugin
  Eigen3::Eigen
)

# === INSTALLATION RULES ===
install(
  TARGETS auv_dynamics_plugin
  DESTINATION lib/${PROJECT_NAME}
)

# Install the include directory so other packages could use your headers
install(
  DIRECTORY include/
  DESTINATION include
)

install(DIRECTORY
  launch
  urdf
  worlds
  config
  models
  DESTINATION share/${PROJECT_NAME}/
)

install(
  PROGRAMS scripts/keyboard_teleop.py
  DESTINATION lib/${PROJECT_NAME}
)

# Final command
ament_package()
